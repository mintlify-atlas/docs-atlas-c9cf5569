## Function signature

```typescript
interface Query {
  categoryId?: string;
  colorId?: string;
  sizeId?: string;
  isFeatured?: boolean;
}

const getProducts = async ({
  categoryId,
  colorId,
  sizeId,
  isFeatured,
}: Query): Promise<Product[]>
```

## HTTP request

```bash
GET ${NEXT_PUBLIC_API_URL}/products
GET ${NEXT_PUBLIC_API_URL}/products?categoryId=<id>&colorId=<id>&sizeId=<id>&isFeatured=true
```

Query parameters are appended only when provided. Multiple filters are joined with `&`.

## Parameters

<ParamField query="categoryId" type="string">
  Filter products by category ID. When omitted, products from all categories are returned.
</ParamField>

<ParamField query="colorId" type="string">

</ParamField>

## Source

```typescript actions/get-products.tsx
import { Product } from "@/types/types";

const URL = `${process.env.NEXT_PUBLIC_API_URL}/products`;

interface Query {
  categoryId?: string;
  colorId?: string;
  sizeId?: string;
  isFeatured?: boolean;
}

const getProducts = async ({
  categoryId,
  colorId,
  sizeId,
  isFeatured,
}: Query): Promise<Product[]> => {
  const queryParams = [];

  if (categoryId) queryParams.push(`categoryId=${categoryId}`);
  if (colorId) queryParams.push(`colorId=${colorId}`);
  if (sizeId) queryParams.push(`sizeId=${sizeId}`);
  if (isFeatured !== undefined) queryParams.push(`isFeatured=${isFeatured}`);

  const urlWithQuery =
    queryParams.length > 0 ? `${URL}?${queryParams.join("&")}` : URL;

  const res = await fetch(urlWithQuery);
  return res.json();
};

export default getProducts;
```

## Usage examples

### Homepage — featured products only

```tsx app/(routes)/page.tsx
import getProducts from "@/actions/get-products";

const HomePage = async () => {
  const products = await getProducts({ isFeatured: true });

  return (
    <ProductList title="Featured Products" items={products} />
  );
};
```

### Category page — filtered by category, size, and color

```tsx app/(routes)/category/[categoryId]/page.tsx
import getProducts from "@/actions/get-products";

const CategoryPage = async ({ params, searchParams }: CategoryProps) => {
  const products = await getProducts({
    categoryId: params.categoryId,
    colorId: searchParams.colorId,
    sizeId: searchParams.sizeId,
  });

  return <ProductList title="Products" items={products} />;
};
```
